[docs]def calculate_X(i, j, p, h,side): #这里有计算冗余
    if side == 'l':
        if p[i] > p[j]:
            return (h[j] - h[i]) / (p[i] - p[j])
        else:
            return float('inf')
    else:
        if p[i] < p[j]:
            return (h[j] - h[i]) / (p[i] - p[j])
        else:
            return -float('inf')




[docs]def calculate_X_j(j, p, h, side, need=0):
    if side == 'l':
        next = 0
        min_X = float('inf')
        for i in range(len(p)):
            X_i_j = calculate_X(i, j, p, h, side)
            if X_i_j <= min_X:
                min_X = X_i_j
                next=i
        if need:
            return min_X,next
        else:
            return min_X
    else:
        next = 0
        max_X = -float('inf')
        for i in range(len(p)):
            X_i_j = calculate_X(i, j, p, h, side)
            if X_i_j >= max_X:
                max_X =  X_i_j
                next=i
        if need:
            return max_X, next
        else:
            return max_X




[docs]def found0(n_side, S, h,bound):
    Xf1,nextf = calculate_X_j(n_side[0], S[:n_side[0] + 1], h[:n_side[0] + 1], 'r',need=1)
    Xz1,nextz = calculate_X_j(0, S[- n_side[1] - 1:], h[- n_side[1] - 1:], 'l',need=1)
    if Xf1 <= Xz1:
        X0 = [max(min(Xf1, bound[1]), bound[0]), max(min(Xz1, bound[1]), bound[0])]
    else:
        C0= (h[nextz+n_side[0]] - h[nextf]) / (S[nextf] - S[nextz+n_side[0]])
        C0= max(min(C0, bound[1]), bound[0])
        X0 = [C0, C0]
    return X0




[docs]def calculate_oneside(C,bound,n_side, S, h,side): #bound=[alpha,beta],n_side=[n_L,n_R]
    X0 = found0(n_side, S, h, bound)
    if side == 'l':
        S= S[:n_side[0] + 1]
        h = h[:n_side[0] + 1]
        C.append(bound[0])  # C is a list
        stack = []
        for i in range(n_side[0] ):
            if len(stack) == 0:
                X_n = calculate_X_j(i, S, h,side)
                C.append(max(min(X_n,X0[0]),bound[0]))
                stack.append(i)
                continue
            if i == n_side[0]-1:
                C.append(X0[0])
                break
            X_i = calculate_X_j(i, S, h,side)
            s = abs(stack[-1])
            X_s = calculate_X_j(s, S, h,side)
            if X_i <= X_s:
                C.append(min(max(X_s, bound[0]),X0[0]))
                stack.append(s)
                continue
            else:
                C.append(min(max(X_i, bound[0]), X0[0]))
                stack.append(i)
    else :
        S = S[- n_side[1] - 1:]
        h = h[- n_side[1] - 1:]
        C.append(bound[1])  # C is a list
        stack = []
        for i in range(n_side[1], 0, -1):
            if i == 1:
                C.append(X0[1])
                break
            if len(stack) == 0:
                X_n = calculate_X_j(i, S, h,side)
                C.append(min(max(X_n,X0[1]),bound[1]))
                stack.append(i)
                continue
            X_i = calculate_X_j(i, S, h,side)
            s = abs(stack[-1])
            X_s = calculate_X_j(s, S, h,side)
            if X_i >= X_s:
                C.append(max(min(X_s, bound[1]),X0[1]))
                stack.append(s)
            else:
                C.append(max(min(X_i, bound[1]), X0[1]))
                stack.append(i)




[docs]def calculate_C1(n_R, n_L, alpha, beta, S, h):
    C = [[],[]]
    C[0].append(calculate_oneside(C[0],[alpha,beta],[n_L,n_R], S, h,'l')) #left
    C[0]=C[0][:-1]
    C[1].append(calculate_oneside(C[1],[alpha,beta],[n_L,n_R], S, h,'r')) #right
    C[1]=C[1][::-1][1:]
    C = C[0] + C[1]
    C = torch.tensor(C).cuda()
    return C




[docs]def calculate_C(n_R, n_L, alpha, beta, S, h):
    C = torch.zeros(len(S)+1)
    C[0]=alpha
    for i in range(1,len(h)):
        C[i]=(h[i-1]-h[i])/(S[i]-S[i-1])
    C[len(h)]=beta
    return C

[docs]def h_initial():
    divsion = np.zeros(n_R+n_L)
    for i in range(n_R+n_L):
        divsion[i] = alpha + (i+1)*(beta-alpha)/(n_R+n_L+1)
    A= np.diag(np.ones(n_R+n_L+1)) - np.diag(np.ones(n_R+n_L), k=1)
    h_initial, residuals, _, _ = np.linalg.lstsq(A[:-1], divsion, rcond=None)
    #h_initial = [1,1,13,-1,13,14,5,-6,6,3,1,-4,15,2,6]
    return h_initial
