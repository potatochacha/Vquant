[docs]def compute_gradient(nu, C, S, uniw, sigmaw, grandtype):
    """
    Compute the gradient of E(h).\n
    grandtype: Parameter, "mixgrand" means considering the cost(h).
    """
    if grandtype == "mixgrad":
        grad = torch.zeros(len(nu),dtype=torch.float64).cuda()
        gupper = mu.pdf(C[-2].item()) * (S[-1].item() - C[-2].item()) ** 2
        glettr = mu.pdf(C[1].item()) * (S[0].item() - C[1].item()) ** 2
        gupper = gupper / (S[-1].item() - S[-2].item())
        glettr = -glettr / (S[0].item() - S[1].item())
        for i in range(len(nu)):  # n_R+n_L = len(C)-2; n=n_R+n_L+1=len(C)-1
            upper = mu.cdf(C[i + 1].item())
            lettr = mu.cdf(C[i].item())
            grad[i] = -nu[i] + upper - lettr
        grad[0] = grad[0] + glettr
        grad[-1] = grad[-1] + gupper
    else:
        grad=torch.zeros(len(nu),dtype=torch.float64).cuda()
        g=torch.zeros(len(nu),dtype=torch.float64).cuda()
        for i in range(len(nu)): # n_R+n_L = len(C)-2; n=n_R+n_L+1=len(C)-1
            upper = mu.cdf(C[i+1].item())
            lettr = mu.cdf(C[i].item())
            grad[i] = -nu[i]+upper-lettr
            g[i] = upper-lettr
    return grad




[docs]def Eh(C, S, h, uniw, sigmaw, nu):
    """
    Compute the number of E(h).
    """
    iE = torch.zeros(len(S),dtype=torch.float64).cuda()
    for i in range(len(S)):
        le = C[i].item() * S[i].item() + h[i].item()
        ue = C[i+1].item() * S[i].item() + h[i].item()
        iE[i] = mu.expect(func=lambda x: x * S[i].item() + h[i].item() , ub=C[i + 1].item(), lb=C[i].item())
    # A = sum(iE)
    # B= torch.dot(nu, h)
    E = sum(iE) - torch.dot(nu, h)
    return E

[docs]def G(x,C, h, uniw, sigmaw, alpha, beta, n_R, n_L, S, distr):
    """
    Implement nu = G(h).\n
    distr: Parameter, deside the distribution of nu.
    """
    # W_k = (C[k+n^L],C[k+n^L+1]),n_R+n_L = len(C)
    U = mu.cdf(C[-1].item()) - mu.cdf(C[0].item())
    nu = torch.zeros(len(h),dtype=torch.float64).cuda()
    if distr=='neardist':
        for i in range(1, len(h)-1):
            nu[i] = 1/(beta-alpha)
        nu[0] = (S[0] - alpha + 0.5) / (beta - alpha)
        nu[-1] = 1-sum(ele for ele in nu)
        nu = U * nu
    elif distr=='uniform':
        for i in range(len(h)-1):
            nu[i] = (C[i+1]-C[i])/(beta-alpha)
        nu[-1] = 1-sum(ele for ele in nu)
        nu = U * nu
    elif distr=='random':
        for i in range(len(h)-1):
            nu[i] = 1/(n_L+n_R+1)
        nu[-1] = 1-sum(ele for ele in nu)
        nu = U * nu
    elif distr=='Binomial':
        n = len(h)
        p = (uniw-S[0].item()) / n
        for i in range(len(h)-1):
            nu[i] = binom.pmf(i,n,p)
        nu[-1] = 1-sum(ele for ele in nu)
        nu = U * nu
    else:
        m = torch.zeros(len(h),dtype=torch.float64).cuda()
        M = torch.sum((x >= C[0]) & (x <= C[-1])).cuda()
        for i in range(len(h)-1):
            m[i] = torch.sum((x >= C[i]) & (x < C[i+1]))
        m[-1] = torch.sum((x >= C[-2]) & (x <= C[-1]))
        nu = m / M
        nu = U * nu
    sigmanu = sum(ele for ele in nu)
    if abs(sigmanu - U) > 0.0001:
        raise ValueError("sigmanu != U")
    return nu

[docs]def h_initial(n_R, n_L, alpha, beta):
    """
    Initial the h^0 at the outset of optimization using an equation-solving approach.
    """
    divsion = np.zeros(n_R+n_L)
    for i in range(n_R+n_L): # 均匀
         divsion[i] = alpha + (i+1)*(beta-alpha)/(n_R+n_L+1)
    eps = np.random.uniform(-0.2, 0.2, size=divsion.shape)
    # divsion = divsion + eps
    A= np.diag(np.ones(n_R+n_L+1)) - np.diag(np.ones(n_R+n_L), k=1)
    h_initial, _, _, _ = np.linalg.lstsq(A[:-1], divsion, rcond=None)
    h_initial = torch.tensor(h_initial, dtype= torch.float64).cuda()
    return h_initial




[docs]def vround(x, C, S):
    """
    Obtain the integer value of x based on interval partition C.\n
    C: interval partitioning.\n
    S: The integer value {p_i}_{i=1}^n.
    """
    x_q = torch.zeros(len(x)).cuda()
    x = x.cpu()
    C = C.cpu()
    # labels = torch.searchsorted(C, x, right=False) #(alpha,]
    labels = np.digitize(x, C, right=True) #(alpha,]
    labels = labels - 1
    for i in range(len(x)):
        if labels[i] == -1:
            labels[i] = 0
        if labels[i] == len(S):
            labels[i] = len(S)-1
        x_q[i] = S[labels[i]]
    return x_q




[docs]def cost(C, S, uniw, sigmaw, x, costtype):
    """
    Compute the cost(h).\n
    costtype: Parameter, deside what kind of cost to use.
    """
    # C = C[1:-1]
    # S = S[1:-1]
    icost = torch.zeros(len(S),dtype=torch.float64).cuda()
    if costtype == "W":
        # w = torch.clamp(x, C[0], C[-1])
        w_q = vround(x, C, S)
        cost = torch.norm(x - w_q)
    elif costtype == "X":
        for i in range(len(S)):
            icost[i] = mu.expect(func=lambda x: (S[i] - x) ** 2, loc=0,
                                              scale=1, ub=C[i + 1], lb=C[i])
        cost = sum(icost)
    elif costtype == "Wnearcost":
        w_q = x.round()
        cost = torch.norm(x - w_q)
    elif costtype == 'Wd':
        x_q = vround(x, C, S)
        wasserstein_distance(x.cpu(),x_q.cpu())
    else:
        for i in range(len(S)):
            icost[i] = mu.expect(func=lambda x: (S[i].item() - x) ** 2,ub=C[i + 1].item(), lb=C[i].item())
        cost = sum(icost)
    return cost

[docs]def lbfgs_optimizer(x, uniw, sigmaw, n_R, n_L, alpha, beta, P, max_iterations=100, tolerance=1e-6):
    """
    Another optimizer use L-BFGS Methond(not suggested).
    """
    h = h_initial(n_R, n_L, alpha, beta)
    C = calculate_C1(n_R, n_L, alpha, beta, P, h)
    nu = G(x, C, h, uniw, sigmaw, alpha, beta, n_R, n_L, P, 'Binomial')
    g = compute_gradient(nu, C, P, uniw, sigmaw, '')
    m = 10  # 历史信息数量，可以根据问题设置合适的值
    S = []
    Y = []
    H0 = 1.0

    for iteration in range(max_iterations):
        if torch.norm(g).item() < tolerance:
            break

        if iteration > 0:
            rho_k = 1.0 / torch.dot(Y[-1], S[-1])
            alpha = []

            q = g
            for i in range(len(S) - 1, -1, -1):
                alpha_k = rho_k * torch.dot(S[i], q)
                alpha.append(alpha_k)
                q -= alpha_k * Y[i]

            r = H0 * q

            for i in range(len(S)):
                beta = rho_k * torch.dot(Y[i], r)
                r += S[i] * (alpha[-1] - beta)
                alpha.pop()

        else:
            r = H0 * g

        d = -r
        alpha = 0.0001

        E2 = Eh(C, P, h + alpha * d, uniw, sigmaw, nu).item()
        E1 = Eh(C, P, h, uniw, sigmaw, nu).item()

        while E2 > E1 + 0.1 * alpha * torch.dot(g, d):
            alpha *= 0.5

        h_new = h + alpha * d
        s = h_new - h
        C = calculate_C1(n_R, n_L, alpha, beta, P, h_new)
        gnew = compute_gradient(nu, C, P, uniw, sigmaw, '')
        y = gnew - g
        H0 = torch.dot(s, y) / torch.dot(y, y)
        g = gnew

        if len(S) == m:
            S.pop(0)
            Y.pop(0)

        S.append(s)
        Y.append(y)
        h = h_new

    return h

    path='vquant_optimize_log'
    if not os.path.isdir(path):
        os.mkdir(path)
    path = os.path.join(path, args.model+"_"+args.dataset)
    if not os.path.isdir(path):
        os.mkdir(path)
    pathname = args.mode + '_W' + str(args.wbit) + 'A' + str(args.wbit)
    num = int(uuid.uuid4().hex[0:4], 16)
    pathname += '_' + str(num)
    path = os.path.join(path, pathname)
    if not os.path.isdir(path):
        os.mkdir(path)    
    return path
