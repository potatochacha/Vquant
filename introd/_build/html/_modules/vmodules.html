    def __init__(self, mode="vquant", bit=8, bound=(-129.5, 128.5), is_signed=True, is_enable=False, is_input=False, args=None, operator=None):
        super(Quantizer, self).__init__()
        self.mode = mode
        self.register_buffer('bit', torch.tensor(1))
        self.bit = torch.tensor(bit)
        self.is_signed = is_signed
        self.is_enable = is_enable
        self.is_enable_input = is_enable
        self.is_input = is_input
        self.args = args

        self.sigmaw = 6

        self.percent = self.args.percent / 100
        self.is_sigma = False

            self.percent = args.sigma
            self.is_sigma = True

        self.cr = args.constraint_radius
        self.pe = args.packed_element

        self.name = None
        self.has_zero = True
        self.quant_weight_tensor = None
        self.register_buffer('x_max', torch.tensor(1.0))
        self.register_buffer('x_min', torch.tensor(1.0))
        self.has_inited_quant_para = False
        if args.xmode == 'vquant':
            self.has_quant_weight = False
        else:
            self.has_quant_weight = True

        self.is_perchannel = True
        if is_input:
            # Input shouldn't be per-channel quantizaton！
            self.is_perchannel = False

        self.tensor_sum = None
        self.tensor_sum_cov = None

[docs]    def adaptive_round(self, x, n_min=None, n_max=None):  # [t_min,t_max]
        # param
        n_L = - n_min.item()
        n_R = n_max.item()
        alpha = x.min().item() - self.bound[0]
        beta = x.max().item() + self.bound[1]
        # alpha = n_min.item() - self.bound[0]
        # beta = n_max.item() + self.bound[1]
        args = self.args
        learning_rate = args.learning_rate
        rate = args.rate  # learning_rate*(1-rate[0])**rate[1] , if vcost_before - vcost < rate[2]
        epsilon = args.epsilon
        grandtype = args.grandtype
        costtype = args.costtype  # cost 'W','X'
        nudistr = args.nudistr  # exp 'neardist','random','uniform'

        uniw = x.mean().item()
        sigmaw = x.std().item()
        testnametime = datetime.now().strftime("%m-%d-%H-%M-%S")

        testname = f"{testnametime}_wquant_{costtype}_{grandtype}_nu={nudistr}_Omega={[alpha, beta]}P={uniw, sigmaw}n={[n_L, n_R]}_lr={learning_rate}{rate}"

        # Get the rounding integer and fraction.
        S = torch.arange(-n_L, n_R + 1).cuda()
        h_optimal, nu, C_optimal = optimize(n_R, n_L, alpha, beta, learning_rate, epsilon, S, x, nudistr, rate,
                                            grandtype, costtype, testname, uniw, sigmaw, self.args)
        # h_optimal = lbfgs_optimizer(x, uniw, sigmaw, n_R, n_L, alpha, beta, S, max_iterations=100, tolerance=1e-6)
        rounding_number = vround(x, C_optimal, S)
        rounding_number = torch.clamp(rounding_number, n_min, n_max)
        assert (rounding_number.unique().numel() <= 2 ** self.bit.item())
        return rounding_number


    @torch.no_grad()
    def _init_quant_para(self, data):
        if self.has_inited_quant_para == False:
            logger.info("QUANT %d bit: %s " % (self.bit.item(), self.name))
            self.updata_packed_element(data)
            self.updata_signed(data)

            x_max = data.max()
            x_min = data.min()
            alpha = self.percent * data.abs().max()
            if self.is_sigma:
                sigma = self._sigma(data)
                alpha = self.percent * sigma
                if self.is_signed:
                    # We also consider the signed activation. Other framworks will skip this tensor.
                    alpha = self.percent * sigma / 1.25

                # For a higher bit-width, using a wider range still will not cause accuracy loss.
                if self.bit < 6:
                    # For small bit, need clip.
                    alpha = min(alpha, x_max)

            def _quant(tensor):
                if self.is_perchannel:
                    x_max = tensor.view(tensor.shape[0], -1).max(1).values
                    x_max = x_max.unsqueeze(1)
                    x_min = tensor.view(tensor.shape[0], -1).min(1).values
                    x_min = x_min.unsqueeze(1)
                else:
                    x_max = tensor.max()
                    x_min = tensor.min()

                scale, zero_point = asymmetric_linear_quantization_params(self.bit, x_min, x_max)
                quant_tensor = linear_quantize(tensor, scale, zero_point, inplace=False)

                n = 2 ** (self.bit - 1)

                if self.mode == "vquant":
                    quantshape = quant_tensor.size()
                    # round_tensor = torch.zeros(quantshape)
                    # for i in range(quantshape[0]):
                    #     channel_tensor = self.adaptive_round(quant_tensor[i].view(-1), -n, n - 1)
                    #     asc = channel_tensor.reshape(quantshape[1:])
                    #     round_tensor[i] = channel_tensor.reshape(quantshape[1:])
                    # quant_tensor = round_tensor
                    # quant_tensor.cpu()
                    round_tensor = self.adaptive_round(quant_tensor.view(-1), n_min =-n, n_max = n - 1)
                    quant_tensor = round_tensor.view(quantshape)
                else:
                    quant_tensor = quant_tensor.round()
                    raise RuntimeError("Unsupported mode: " + self.mode)

                quant_tensor = torch.clamp(quant_tensor, -n, n - 1)  # [-128,127]
                quant_tensor = linear_dequantize(quant_tensor, scale, zero_point, inplace=False)
                return quant_tensor

            if not self.is_input:
                # Weight quantization
                start = time.perf_counter()
                self.quant_weight_tensor = _quant(data) #调用权重的量化_quant
                elapsed = (time.perf_counter() - start)
                logger.info("Quantzation time: %f ms" % (elapsed * 1000))
            else:
                # Activation quantization
                # min
                if self.is_signed:
                    self.x_min = -alpha
                else:
                    self.x_min.data = torch.zeros_like(alpha)
                # max
                self.x_max.data = alpha

        self.has_inited_quant_para = True

    def _forward(self, data):
        tensor, has_quant_weight = AsymmetricQuantFunction.apply(data, self.bit, self.x_min, self.x_max, self.args, self.has_quant_weight)
        self.has_quant_weight = has_quant_weight
        return tensor


[docs]    def tensor_forward(self, tensor, image_size):
        self.image_size = image_size
        if self.mode == "base":
            return tensor
        if not self.is_enable:
            return tensor
        if self.is_input: #判断是否为激活
            if not self.is_enable_input: #如果不是可用激活这里就返回
                return tensor 

        with torch.no_grad():

            self._init_quant_para(tensor)  #生成量化后的权重
            if self.is_input:  #是激活则调用激活的量化_forward量化
                return self._forward(tensor)
            else:  #是权重则调用权重的量化
                return self.quant_weight_tensor

    """

    """

    def __init__(self, mode=None, wbit=None, abit=None, bound=None, args=None):
        """
        weight: bit-setting for weight \n
        full_precision_flag: full precision or not \n
        running_stat: determines whether the activation range is updated or froze
        """
        super(LinearQuantizer, self).__init__()
        assert mode is not None, 'Quantizer is not initilized!'
        self.quant_input = TensorQuantizer(mode=mode, bit=abit, bound=bound, is_signed=False, is_enable=True, args=args,

        self.quant_weight = TensorQuantizer(mode=mode, bit=wbit, bound=bound, is_signed=True, is_enable=True, args=args)
